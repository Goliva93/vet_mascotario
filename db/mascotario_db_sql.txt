CREATE TABLE dbo.Appointment (
    AppointmentId INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    BranchId INT NOT NULL,

    OwnerId INT NOT NULL,
    PetId INT NOT NULL,

    AssignedGroomerUserId INT NULL,

    StartAt DATETIME2 NOT NULL,
    EndAt DATETIME2 NOT NULL,

    Status NVARCHAR(20) NOT NULL CONSTRAINT DF_Appointment_Status DEFAULT ('CONFIRMED'),
    Notes NVARCHAR(500) NULL,

    RescheduledFromAppointmentId INT NULL,
    CancelReason NVARCHAR(200) NULL,
    CancelledAt DATETIME2 NULL,
    CancelledByUserId INT NULL,

    CreatedByUserId INT NULL,
    CreatedAt DATETIME2 NOT NULL CONSTRAINT DF_Appointment_CreatedAt DEFAULT (SYSUTCDATETIME()),
    UpdatedAt DATETIME2 NULL,

    CONSTRAINT FK_Appointment_Branch FOREIGN KEY (BranchId) REFERENCES dbo.Branch(BranchId),
    CONSTRAINT FK_Appointment_Type FOREIGN KEY (AppointmentTypeId) REFERENCES dbo.AppointmentType(AppointmentTypeId),
    CONSTRAINT FK_Appointment_Channel FOREIGN KEY (ChannelId) REFERENCES dbo.Channel(ChannelId),
    CONSTRAINT FK_Appointment_Owner FOREIGN KEY (OwnerId) REFERENCES dbo.Owner(OwnerId),
    CONSTRAINT FK_Appointment_Pet FOREIGN KEY (PetId) REFERENCES dbo.Pet(PetId),
    CONSTRAINT FK_Appointment_AssignedVet FOREIGN KEY (AssignedVetUserId) REFERENCES dbo.[User](UserId),
    CONSTRAINT FK_Appointment_AssignedGroomer FOREIGN KEY (AssignedGroomerUserId) REFERENCES dbo.[User](UserId),
    CONSTRAINT FK_Appointment_RescheduledFrom FOREIGN KEY (RescheduledFromAppointmentId) REFERENCES dbo.Appointment(AppointmentId),
    CONSTRAINT FK_Appointment_CancelledBy FOREIGN KEY (CancelledByUserId) REFERENCES dbo.[User](UserId),
    CONSTRAINT FK_Appointment_CreatedBy FOREIGN KEY (CreatedByUserId) REFERENCES dbo.[User](UserId),
    CONSTRAINT CK_Appointment_Time CHECK (EndAt > StartAt)
);
GO
